## Task 1: Priority derived from phase (E) — TDD

**Files:** `skills/build-project-model/scripts/planning-model.mjs`, `tests/planning-model.test.mjs`

- [ ] **Step 1: Add failing tests** (append to `tests/planning-model.test.mjs`):

```js
import { derivePriority, PHASE_PRIORITY } from '../skills/build-project-model/scripts/planning-model.mjs';

test('derivePriority: maps phase to Jira priority', () => {
  assert.equal(derivePriority('MVP'), 'High');
  assert.equal(derivePriority('Production Ready'), 'Medium');
  assert.equal(derivePriority('Public Release'), 'Low');
  assert.equal(derivePriority('Enterprise'), 'Lowest');
  assert.equal(derivePriority('Nonsense'), 'Medium');
});

test('normalizeModel: stamps priority on epics (from phase) and stories (inherit epic)', () => {
  const model = {
    schemaVersion: '1.0',
    knowledgeModel: { domains: [], architecture: [], techDebt: [], infrastructure: [], security: [], risks: [] },
    planningModel: { phases: [], items: [] },
    backlog: { epics: [{
      id: 'epic-cal', trackerKey: null, phase: 'MVP', type: 'feature', title: 'Calendar',
      stories: [{ id: 's1', trackerKey: null, title: 'Day', acceptanceCriteria: [{ given: 'g', when: 'w', then: 't' }], definitionOfDone: ['code'], tasks: [] }]
    }] }
  };
  const n = normalizeModel(model);
  const epic = n.backlog.epics.find(e => e.id === 'epic-cal');
  assert.equal(epic.priority, 'High');
  assert.equal(epic.stories[0].priority, 'High');
});
```

- [ ] **Step 2: Run** `node --test tests/planning-model.test.mjs` → FAIL (derivePriority not exported).

- [ ] **Step 3: Implement** in `planning-model.mjs` — add near the top:

```js
export const PHASE_PRIORITY = {
  'MVP': 'High',
  'Production Ready': 'Medium',
  'Public Release': 'Low',
  'Scaling': 'Low',
  'Enterprise': 'Lowest',
  'AI': 'Lowest'
};
export function derivePriority(phase) {
  return PHASE_PRIORITY[phase] || 'Medium';
}
```

And in `normalizeModel`, after `ensureDedicatedEpics`/`applyDefaultDoD`, stamp priority:

```js
  for (const epic of m.backlog.epics) {
    epic.priority = derivePriority(epic.phase);
    for (const story of epic.stories || []) story.priority = epic.priority;
  }
```

- [ ] **Step 4: Run** `node --test tests/planning-model.test.mjs` → PASS.

- [ ] **Step 5: Commit** `feat(build): derive Jira priority from phase (E)`.
